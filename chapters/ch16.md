# 챕터 16: 시스템 모니터링과 로그

> 서버 상태를 실시간으로 파악하고 문제를 조기에 발견한다

## 학습 목표
- 시스템 리소스를 다양한 도구로 모니터링할 수 있다
- systemd 저널과 syslog를 분석할 수 있다
- 성능 병목점을 찾고 진단할 수 있다
- 모니터링 자동화를 구축할 수 있다

## 16.1 시스템 리소스 모니터링

### CPU 모니터링

```bash
# 실시간 CPU 사용률
top
htop

# CPU 정보
lscpu
cat /proc/cpuinfo | grep "model name" | head -1
nproc                    # 코어 수

# vmstat - 시스템 통계
vmstat 1 5               # 1초 간격, 5회
vmstat -s                # 메모리 통계 요약

# mpstat - CPU별 사용률 (sysstat 패키지)
sudo apt install sysstat
mpstat -P ALL 1 3
mpstat 1                 # 전체 CPU 합산

# sar - 역사적 데이터 수집/분석
sar -u 1 10              # CPU 사용률 (1초 간격, 10회)
sar -u -f /var/log/sysstat/sa$(date +%d)  # 오늘 히스토리
```

### 메모리 모니터링

```bash
# 메모리 사용량
free -h
free -m

# 상세 메모리 정보
cat /proc/meminfo

# 메모리 사용 프로세스 (상위 10개)
ps aux --sort=-%mem | head -11

# 캐시와 버퍼 확인
vmstat -s | grep -E "used|free|buffer|cache"

# 스왑 사용량
swapon --show
cat /proc/swaps
```

```
free -h 출력 해석:
              total   used   free  shared  buff/cache  available
Mem:           15G    8.2G   3.1G    512M        3.7G       6.5G
Swap:           2G    256M   1.7G

available = 실제 사용 가능한 메모리 (buff/cache 포함)
```

### 디스크 I/O 모니터링

```bash
# iostat - 디스크 I/O 통계
iostat -x 1 5             # 확장 통계, 1초 간격, 5회
iostat -d /dev/sda 1      # 특정 디스크

# iotop - 프로세스별 I/O
sudo apt install iotop
sudo iotop
sudo iotop -o             # I/O 발생 프로세스만

# lsof - 열린 파일 목록
lsof +D /var/log          # 특정 디렉토리
lsof -p 1234              # 특정 PID
lsof -u student           # 특정 사용자
lsof -i :80               # 특정 포트

# 디스크 읽기/쓰기 속도 테스트
dd if=/dev/zero of=/tmp/testfile bs=1M count=100
dd if=/tmp/testfile of=/dev/null bs=1M
```

### 네트워크 모니터링

```bash
# 네트워크 통계
ss -s
netstat -s 2>/dev/null | head -20

# 인터페이스 통계
cat /proc/net/dev
ip -s link

# 대역폭 모니터링
sudo apt install iftop nload
sudo iftop -i ens33
sudo nload ens33

# 연결 수 확인
ss -ant | awk '{print $1}' | sort | uniq -c
ss -s

# tcpdump 패킷 캡처
sudo tcpdump -i ens33 -n -c 100
```

## 16.2 로그 시스템 이해

### 로그 파일 위치

```bash
/var/log/
├── syslog          # 일반 시스템 로그 (Ubuntu 기본)
├── auth.log        # 인증/보안 로그 (로그인, sudo)
├── kern.log        # 커널 로그
├── dmesg           # 부팅 시 커널 메시지
├── dpkg.log        # 패키지 설치/제거
├── apt/            # APT 로그
├── nginx/          # Nginx 액세스/에러
│   ├── access.log
│   └── error.log
├── mysql/          # MySQL 로그
├── fail2ban.log    # fail2ban 로그
└── ufw.log         # 방화벽 로그
```

### 로그 레벨

```
emerg   (0) - 시스템 사용 불가
alert   (1) - 즉시 조치 필요
crit    (2) - 심각한 오류
err     (3) - 오류
warning (4) - 경고
notice  (5) - 정상이지만 중요
info    (6) - 정보
debug   (7) - 디버그 정보
```

## 16.3 journalctl - systemd 저널

```bash
# 전체 로그
journalctl

# 최신 로그 (tail)
journalctl -f
journalctl -n 50          # 최근 50줄

# 특정 서비스 로그
journalctl -u nginx
journalctl -u nginx -f    # 실시간

# 시간 범위 필터
journalctl --since "2024-01-15 10:00:00"
journalctl --since "1 hour ago"
journalctl --since yesterday
journalctl --since "2024-01-01" --until "2024-01-31"

# 부팅별 로그
journalctl -b             # 현재 부팅
journalctl -b -1          # 이전 부팅
journalctl --list-boots   # 부팅 목록

# 우선순위별 필터
journalctl -p err         # 오류 이상만
journalctl -p warning     # 경고 이상만

# 특정 PID
journalctl _PID=1234

# JSON 출력
journalctl -o json-pretty | head -50

# 저널 크기 관리
journalctl --disk-usage
sudo journalctl --vacuum-size=500M   # 500MB로 줄이기
sudo journalctl --vacuum-time=30d    # 30일 이전 삭제
```

## 16.4 rsyslog 설정

```bash
# rsyslog 설정
sudo nano /etc/rsyslog.conf

# 원격 로그 서버로 전송
# /etc/rsyslog.d/remote.conf
*.* @logserver.example.com:514    # UDP
*.* @@logserver.example.com:514   # TCP

# 특정 앱 로그 분리
local1.* /var/log/myapp.log

# rsyslog 재시작
sudo systemctl restart rsyslog
```

## 16.5 성능 분석 도구

### perf - Linux 성능 분석

```bash
# 설치
sudo apt install linux-tools-generic linux-tools-$(uname -r)

# CPU 성능 이벤트 수집
sudo perf stat ls

# 특정 프로세스 프로파일링
sudo perf top -p 1234

# 시스템 전체 프로파일링
sudo perf record -g -a sleep 10
sudo perf report
```

### strace - 시스템 콜 추적

```bash
# 프로세스 시스템 콜 추적
strace ls /tmp

# 실행 중인 프로세스 추적
sudo strace -p 1234

# 요약 통계
strace -c ls /tmp
```

### lsof - 열린 파일 및 네트워크

```bash
# 모든 열린 파일
lsof | head -20

# 특정 포트 사용 프로세스
lsof -i :80
lsof -i :22
lsof -i TCP:443

# 삭제됐지만 열려있는 파일 (디스크 공간 회복 안 되는 경우)
lsof +L1
```

## 16.6 모니터링 자동화

### 간단한 모니터링 스크립트

```bash
#!/bin/bash
# /opt/scripts/system_monitor.sh

ALERT_EMAIL="admin@example.com"
CPU_WARN=80
MEM_WARN=85
DISK_WARN=90

send_alert() {
    echo "$2" | mail -s "[서버 경고] $1" "$ALERT_EMAIL"
}

# CPU 체크
CPU=$(top -bn1 | grep "Cpu(s)" | awk '{print $2}' | tr -d '%us,')
[ "${CPU%.*}" -gt $CPU_WARN ] && send_alert "CPU 높음" "CPU: ${CPU}%"

# 메모리 체크
MEM=$(free | awk '/Mem/{printf "%.0f", $3/$2*100}')
[ "$MEM" -gt $MEM_WARN ] && send_alert "메모리 높음" "Memory: ${MEM}%"

# 디스크 체크
df -h | grep -vE '^Filesystem|tmpfs' | while read -r line; do
    DISK=$(echo "$line" | awk '{print $5}' | tr -d '%')
    MOUNT=$(echo "$line" | awk '{print $6}')
    [ "${DISK:-0}" -gt $DISK_WARN ] && \
      send_alert "디스크 높음" "Disk $MOUNT: ${DISK}%"
done

# 서비스 체크
for SVC in nginx mysql ssh; do
    systemctl is-active --quiet "$SVC" || \
      send_alert "서비스 중단" "$SVC 서비스가 중단됨!"
done
```

### Prometheus + Grafana 소개

```bash
# Node Exporter 설치 (시스템 메트릭 수집)
wget https://github.com/prometheus/node_exporter/releases/download/v1.7.0/node_exporter-1.7.0.linux-amd64.tar.gz
tar -xzf node_exporter-*.tar.gz
sudo mv node_exporter-*/node_exporter /usr/local/bin/

# systemd 서비스로 등록
sudo tee /etc/systemd/system/node_exporter.service << EOF
[Unit]
Description=Node Exporter

[Service]
User=nobody
ExecStart=/usr/local/bin/node_exporter

[Install]
WantedBy=multi-user.target
EOF

sudo systemctl enable --now node_exporter

# 메트릭 확인 (포트 9100)
curl http://localhost:9100/metrics | head -20
```

## 핵심 요약

| 도구 | 용도 |
|------|------|
| `top/htop` | 실시간 CPU/메모리/프로세스 |
| `vmstat` | 시스템 성능 통계 |
| `iostat` | 디스크 I/O |
| `iotop` | 프로세스별 디스크 I/O |
| `iftop/nload` | 네트워크 대역폭 |
| `journalctl -u svc` | systemd 서비스 로그 |
| `journalctl -p err` | 오류 로그 필터 |
| `lsof -i :포트` | 포트 사용 프로세스 |

- **`journalctl -f -u 서비스명`**: 실시간 서비스 로그 모니터링 필수 습관
- **`vmstat 1`**: 성능 이슈 발생 시 첫 번째로 확인
- **logrotate**: 로그 파일 관리 자동화
