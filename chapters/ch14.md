## 14.1 스크립트 기초

### 첫 번째 스크립트

```bash
#!/bin/bash
# 첫 줄: shebang (스크립트 인터프리터 지정)
# #! + /bin/bash: Bash로 실행

echo "Hello, Linux!"
echo "현재 날짜: $(date)"
echo "현재 사용자: $USER"
echo "현재 디렉토리: $PWD"
```

```bash
# 실행 권한 부여
chmod +x script.sh

# 실행 방법
./script.sh
bash script.sh
sh script.sh
```

### 스크립트 모범 사례

```bash
#!/bin/bash
set -euo pipefail
# -e: 오류 발생 시 즉시 종료
# -u: 정의되지 않은 변수 사용 시 오류
# -o pipefail: 파이프 중간 오류도 감지

IFS=$'\n\t'  # 안전한 필드 구분자
```

## 14.2 변수

### 변수 선언과 사용

```bash
# 변수 선언 (= 주변에 공백 없음!)
NAME="Ubuntu"
VERSION=22
PI=3.14159

# 사용 ($변수명 또는 ${변수명})
echo $NAME
echo "Version: ${VERSION}"
echo "Pi = ${PI}"

# 읽기 전용 변수
readonly MAX_RETRY=3
declare -r CONFIG_FILE="/etc/app.conf"

# 정수 변수
declare -i count=0
count=count+5        # 산술 연산 자동 처리

# 배열
fruits=("apple" "banana" "cherry")
echo ${fruits[0]}            # apple
echo ${fruits[@]}            # 모든 요소
echo ${#fruits[@]}           # 배열 길이
fruits+=("date")             # 요소 추가

# 연관 배열 (해시맵)
declare -A scores
scores["alice"]=95
scores["bob"]=87
echo ${scores["alice"]}
```

### 특수 변수

```bash
$0       # 스크립트 이름
$1 ~ $9  # 위치 인자
$@       # 모든 인자 (각각 따옴표)
$*       # 모든 인자 (하나의 문자열)
$#       # 인자 수
$?       # 마지막 명령의 종료 코드 (0=성공)
$$       # 현재 스크립트 PID
$!       # 마지막 백그라운드 프로세스 PID
$_       # 마지막 명령의 마지막 인자
```

```bash
#!/bin/bash
echo "스크립트명: $0"
echo "첫 번째 인자: $1"
echo "모든 인자: $@"
echo "인자 수: $#"

if [ $# -lt 2 ]; then
    echo "사용법: $0 <이름> <나이>"
    exit 1
fi
```

### 변수 치환

```bash
NAME=""
echo ${NAME:-"기본값"}         # NAME이 비어있으면 "기본값" 출력
echo ${NAME:="기본값"}         # NAME이 비어있으면 "기본값"으로 설정
echo ${NAME:+"사용됨"}         # NAME이 설정되어 있으면 "사용됨"

FILE="/home/student/docs/file.txt"
echo ${FILE##*/}               # file.txt (마지막 /이후)
echo ${FILE%/*}                # /home/student/docs (마지막 /이전)
echo ${FILE%.txt}              # .txt 제거
echo ${FILE/student/admin}     # 치환
echo ${#FILE}                  # 문자열 길이
```

## 14.3 조건문

### if 문

```bash
# 파일 관련 조건
if [ -f "/etc/passwd" ]; then
    echo "파일 존재"
fi

if [ -d "/tmp" ]; then
    echo "디렉토리 존재"
fi

if [ ! -e "/path/to/file" ]; then
    echo "파일 없음"
fi

# 숫자 비교
if [ $count -gt 10 ]; then
    echo "10보다 큼"
elif [ $count -eq 10 ]; then
    echo "10과 같음"
else
    echo "10보다 작음"
fi

# 문자열 비교
if [ "$NAME" = "Ubuntu" ]; then
    echo "Ubuntu입니다"
fi

if [ -z "$VAR" ]; then
    echo "변수가 비어있음"
fi

if [ -n "$VAR" ]; then
    echo "변수가 비어있지 않음"
fi

# 복합 조건
if [ $count -gt 0 ] && [ $count -lt 100 ]; then
    echo "1~99 사이"
fi

# [[ ]] 확장 테스트 (Bash 전용, 더 강력)
if [[ $NAME == "Ubuntu"* ]]; then
    echo "Ubuntu로 시작"
fi

if [[ $VERSION =~ ^[0-9]+\.[0-9]+$ ]]; then
    echo "유효한 버전 형식"
fi
```

### 파일 테스트 연산자

| 연산자 | 의미 |
|--------|------|
| `-f file` | 일반 파일 존재 |
| `-d dir` | 디렉토리 존재 |
| `-e path` | 경로 존재 (파일/디렉토리) |
| `-r file` | 읽기 가능 |
| `-w file` | 쓰기 가능 |
| `-x file` | 실행 가능 |
| `-s file` | 크기가 0보다 큼 |
| `-L file` | 심볼릭 링크 |
| `f1 -nt f2` | f1이 f2보다 최신 |

### case 문

```bash
case $1 in
    start)
        echo "서비스 시작"
        ;;
    stop)
        echo "서비스 중지"
        ;;
    restart|reload)
        echo "서비스 재시작"
        ;;
    *)

        exit 1
        ;;
esac
```

## 14.4 반복문

### for 문

```bash
# 범위
for i in {1..5}; do
    echo "번호: $i"
done

# 배열
for fruit in "${fruits[@]}"; do
    echo "과일: $fruit"
done

# 파일 목록
for file in /var/log/*.log; do
    echo "로그: $file"
done

# C 스타일
for ((i=0; i<10; i++)); do
    echo $i
done

# 명령어 출력
for user in $(cut -d: -f1 /etc/passwd); do
    echo "사용자: $user"
done
```

### while/until 문

```bash
# while
count=0
while [ $count -lt 5 ]; do
    echo "카운트: $count"
    ((count++))
done

# until (조건이 참이 될 때까지)
until ping -c1 google.com &>/dev/null; do
    echo "네트워크 기다리는 중..."
    sleep 5
done
echo "네트워크 연결됨"

# 파일 라인 읽기
while IFS= read -r line; do
    echo "라인: $line"
done < /etc/hosts

# 파이프로 읽기
ps aux | while read line; do
    echo "$line"
done
```

## 14.5 함수

```bash
#!/bin/bash

# 함수 정의
greet() {
    local name=$1    # local: 함수 내 지역 변수
    local age=${2:-0}
    echo "안녕하세요, ${name}님 (나이: ${age})"
}

# 반환값 (종료 코드)
is_root() {
    if [ $(id -u) -eq 0 ]; then
        return 0   # true
    else
        return 1   # false
    fi
}

# 결과 반환 (echo 사용)
get_disk_usage() {
    local path=${1:-"/"}
    df -h "$path" | awk 'NR==2 {print $5}' | tr -d '%'
}

# 함수 호출
greet "Alice" 25
greet "Bob"

if is_root; then
    echo "root 사용자입니다"
fi

usage=$(get_disk_usage "/")
echo "디스크 사용률: ${usage}%"
```

## 14.7 실전 스크립트

### 시스템 헬스체크 스크립트

```bash
#!/bin/bash
# /opt/scripts/health_check.sh

set -euo pipefail

# 임계값
CPU_THRESHOLD=80
MEM_THRESHOLD=85
DISK_THRESHOLD=90

check_cpu() {
    local cpu=$(top -bn1 | grep "Cpu(s)" | awk '{print $2}' | tr -d '%us,')
    local cpu_int=${cpu%.*}
    if [ "${cpu_int:-0}" -gt $CPU_THRESHOLD ]; then
        echo "WARN: CPU 사용률 높음: ${cpu}%"
        return 1
    fi
    echo "OK: CPU ${cpu}%"
}

check_memory() {
    local mem=$(free | grep Mem | awk '{printf "%.0f", $3/$2*100}')
    if [ "$mem" -gt $MEM_THRESHOLD ]; then
        echo "WARN: 메모리 사용률 높음: ${mem}%"
        return 1
    fi
    echo "OK: Memory ${mem}%"
}

check_disk() {
    while IFS= read -r line; do
        local usage=$(echo "$line" | awk '{print $5}' | tr -d '%')
        local mount=$(echo "$line" | awk '{print $6}')
        if [ "${usage:-0}" -gt $DISK_THRESHOLD ]; then
            echo "WARN: 디스크 사용률 높음 ${mount}: ${usage}%"
        fi
    done < <(df -h | grep -v tmpfs | tail -n +2)
}

check_service() {
    local service=$1
    if systemctl is-active --quiet "$service"; then
        echo "OK: $service 실행 중"
    else
        echo "ERROR: $service 중지됨"
        return 1
    fi
}

echo "=== 시스템 헬스체크 $(date) ==="
check_cpu
check_memory
check_disk
for svc in nginx mysql ssh; do
    check_service "$svc"
done
echo "=== 체크 완료 ==="
```

## 14.8 디버깅

```bash
# 디버그 모드 실행
bash -x script.sh

# 스크립트 내에서 디버그 활성화
set -x    # 실행 명령 출력
set +x    # 디버그 모드 해제

# 구문 검사
bash -n script.sh

# shellcheck 설치 (정적 분석)
sudo apt install shellcheck
shellcheck script.sh
```

## 핵심 요약

- **shebang**: `#!/bin/bash`로 시작, `chmod +x`로 실행 권한
- **set -euo pipefail**: 안전한 스크립트의 시작
- **[ ]**: 기본 테스트, **[[ ]]**: Bash 확장 테스트 (정규식 지원)
- **local**: 함수 내 지역 변수 반드시 사용
- **$?**: 명령 성공 여부 확인 (0=성공)
- **shellcheck**: 스크립트 작성 후 항상 검사
