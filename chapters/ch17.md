## 17.1 Python 개발 환경

### Python 설치와 버전 관리

```bash
# 기본 Python 확인
python3 --version
which python3

# 최신 Python 설치 (deadsnakes PPA)
sudo add-apt-repository ppa:deadsnakes/python
sudo apt update
sudo apt install python3.12 python3.12-venv python3.12-dev

# 여러 버전 관리 (update-alternatives)
sudo update-alternatives --install /usr/bin/python3 python3 /usr/bin/python3.11 1
sudo update-alternatives --install /usr/bin/python3 python3 /usr/bin/python3.12 2
sudo update-alternatives --config python3

# pip 설치 및 업그레이드
sudo apt install python3-pip
pip3 install --upgrade pip
```

### 가상 환경 (venv)

```bash
# 가상 환경 생성
python3 -m venv myproject-env
python3.12 -m venv myproject-env    # 특정 버전

# 활성화
source myproject-env/bin/activate
# (myproject-env) student@ubuntu:~$

# 비활성화
deactivate

# 패키지 설치
pip install flask sqlalchemy

# 설치된 패키지 저장
pip freeze > requirements.txt

# requirements.txt에서 설치
pip install -r requirements.txt
```

### pyenv - 버전 관리자

```bash
# pyenv 설치
curl https://pyenv.run | bash

# .bashrc에 추가
echo 'export PYENV_ROOT="$HOME/.pyenv"' >> ~/.bashrc
echo 'command -v pyenv >/dev/null || export PATH="$PYENV_ROOT/bin:$PATH"' >> ~/.bashrc
echo 'eval "$(pyenv init -)"' >> ~/.bashrc
source ~/.bashrc

# 빌드 의존성 설치
sudo apt install -y build-essential libssl-dev zlib1g-dev \
  libbz2-dev libreadline-dev libsqlite3-dev libncurses5-dev \
  libncursesw5-dev xz-utils tk-dev libffi-dev liblzma-dev

# Python 설치
pyenv install 3.12.0
pyenv install 3.11.5

# 버전 설정
pyenv global 3.12.0         # 전체 기본
pyenv local 3.11.5          # 현재 디렉토리
pyenv shell 3.10.0          # 현재 쉘

# 설치된 버전 목록
pyenv versions
```

## 17.2 Node.js 개발 환경

### nvm - Node 버전 관리자

```bash
# nvm 설치
curl -o- https://raw.githubusercontent.com/nvm-sh/nvm/v0.39.5/install.sh | bash
source ~/.bashrc

# Node.js 설치
nvm install 20              # 최신 LTS
nvm install --lts           # 최신 LTS 버전
nvm install 18.17.0         # 특정 버전

# 버전 전환
nvm use 20
nvm use --lts
nvm alias default 20        # 기본 버전 설정

# 설치된 버전 목록
nvm list
nvm list --no-alias

# 버전 확인
node --version
npm --version
```

### npm 기본 사용법

```bash
# 프로젝트 초기화
mkdir myapp && cd myapp
npm init -y

# 패키지 설치
npm install express         # 프로덕션 의존성
npm install --save-dev jest # 개발 의존성
npm install -g pm2          # 글로벌 설치

# 패키지 업데이트
npm update
npm outdated                # 업데이트 가능한 목록

# 스크립트 실행
npm run start
npm run test
npm run build

# package.json 예시
cat package.json
```

### PM2 - 프로세스 관리

```bash
# PM2 설치
npm install -g pm2

# 앱 시작

pm2 start server.js -i max    # CPU 코어 수만큼 클러스터

# 부팅 시 자동 시작
```

## 17.3 Java 개발 환경

### OpenJDK 설치

```bash
# 사용 가능한 버전 확인
apt search openjdk

# OpenJDK 21 (LTS) 설치
sudo apt install openjdk-21-jdk

# 여러 JDK 관리
sudo update-alternatives --config java
sudo update-alternatives --config javac

# JAVA_HOME 설정
echo 'export JAVA_HOME=$(dirname $(dirname $(readlink -f $(which java))))' >> ~/.bashrc
echo 'export PATH=$JAVA_HOME/bin:$PATH' >> ~/.bashrc
source ~/.bashrc

java -version
javac -version
echo $JAVA_HOME
```

### Maven 설치

```bash
sudo apt install maven
mvn --version

# 프로젝트 생성
mvn archetype:generate \
  -DgroupId=com.example \
  -DartifactId=myapp \
  -DarchetypeArtifactId=maven-archetype-quickstart \
  -DinteractiveMode=false

cd myapp
mvn compile
mvn test
mvn package
java -jar target/myapp-1.0-SNAPSHOT.jar
```

## 17.4 Git 설정과 활용

### 초기 설정

```bash
# 사용자 정보 설정
git config --global user.name "Kim Student"
git config --global user.email "student@example.com"

# 기본 에디터 설정
git config --global core.editor vim
git config --global core.editor "code --wait"  # VS Code

# 기본 브랜치명
git config --global init.defaultBranch main

# 줄 끝 설정
git config --global core.autocrlf input  # Linux/Mac

# 설정 확인
git config --list
cat ~/.gitconfig
```

### .gitignore 설정

```bash
# 글로벌 .gitignore
cat > ~/.gitignore_global << 'EOF'
# OS
.DS_Store
Thumbs.db
*.swp
*.swo

# 환경 파일
.env
.env.local
*.env

# 에디터
.idea/
.vscode/
*.code-workspace
EOF

git config --global core.excludesfile ~/.gitignore_global
```

### 유용한 Git 설정

```bash
# ~/.gitconfig 추가 설정
git config --global alias.st status
git config --global alias.co checkout
git config --global alias.br branch
git config --global alias.lg "log --graph --oneline --all --decorate"
git config --global alias.unstage "reset HEAD --"

# 색상 출력
git config --global color.ui auto

# push 기본 설정
git config --global push.default current

# pull 기본 설정 (rebase)
git config --global pull.rebase true
```

### SSH 키로 GitHub/GitLab 연결

```bash
# SSH 키 생성
ssh-keygen -t ed25519 -C "student@example.com" -f ~/.ssh/github_key

# ssh-agent에 키 추가
eval $(ssh-agent)
ssh-add ~/.ssh/github_key

# ~/.ssh/config 설정
cat >> ~/.ssh/config << 'EOF'
Host github.com
    HostName github.com
    User git
    IdentityFile ~/.ssh/github_key

Host gitlab.com
    HostName gitlab.com
    User git
    IdentityFile ~/.ssh/github_key
EOF

# 공개키를 GitHub/GitLab에 등록
cat ~/.ssh/github_key.pub

# 연결 테스트
ssh -T git@github.com
```

## 17.5 VS Code 원격 개발

### Remote-SSH 설정

```bash
# 로컬 PC에서 VS Code 설치 후
# Extension: Remote - SSH 설치

# ~/.ssh/config (로컬 PC에서)
Host ubuntu-server
    HostName 192.168.1.100
    User student
    IdentityFile ~/.ssh/id_ed25519
    ServerAliveInterval 60
```

```
VS Code에서:
1. Ctrl+Shift+P → "Remote-SSH: Connect to Host"
2. ubuntu-server 선택
3. 연결 후 서버에서 VS Code Server 자동 설치
4. 로컬처럼 원격 파일 편집 가능
```

### .env 파일 관리

```bash
# 프로젝트별 .env 파일
cat > /opt/myapp/.env << 'EOF'
DB_HOST=localhost
DB_PORT=3306
DB_NAME=myapp
DB_USER=appuser
DB_PASSWORD=secure_password
REDIS_URL=redis://localhost:6379
SECRET_KEY=your-secret-key
NODE_ENV=production
PORT=3000
```
